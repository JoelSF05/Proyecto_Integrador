CREATE VIEW public.vista_pagos_trabajadores AS
 SELECT t.id_trab,
    t.nom_trab,
    t.ape_trab,
    t.tipo_pago,
    a.fec_asist,
    a.sacos_cosechados,

  WHERE (t.activo = true);
